/** * ฟังก์ชัน doPost: สำหรับรับข้อมูลจากเว็บแล้วบันทึกลง Sheet (อันเดิมที่คุณมี) */ function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var dateTime = e.parameter.Date_Time; var machineId = e.parameter.Machine_ID; var transactionType = e.parameter.Transaction_Type; var description = e.parameter.Description; var amount = e.parameter.Amount; var notes = e.parameter.Notes; sheet.appendRow([dateTime, machineId, transactionType, description, amount, notes]); return ContentService .createTextOutput(JSON.stringify({ "result": "success", "message": "บันทึกข้อมูลเรียบร้อย" })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ "result": "error", "error": error.message })) .setMimeType(ContentService.MimeType.JSON); } } /** * ฟังก์ชัน doGet (ใหม่): สำหรับดึงข้อมูลจาก Sheet เพื่อส่งไปแสดงในตารางบนเว็บ */ function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // ดึงข้อมูลทั้งหมดใน Sheet var data = sheet.getDataRange().getDisplayValues(); var result = []; // เริ่ม loop จาก i=1 เพื่อข้ามหัวตาราง (แถวที่ 0) for (var i = 1; i < data.length; i++) { // ตรวจสอบว่าแถวนั้นไม่ใช่แถวว่าง if(data[i][0] !== "") { result.push({ "dateTime": data[i][0], "machineId": data[i][1], "type": data[i][2], "description": data[i][3], "amount": data[i][4], "notes": data[i][5] }); } } // ส่งข้อมูลกลับไปในรูปแบบ JSON (และเรียงข้อมูลล่าสุดขึ้นก่อนด้วยการ .reverse()) return ContentService .createTextOutput(JSON.stringify({ "result": "success", "data": result.reverse() })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ "result": "error", "error": error.message })) .setMimeType(ContentService.MimeType.JSON); } }
